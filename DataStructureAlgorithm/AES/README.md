void make_enc(unsigned char *in, unsigned char *out, unsigned char(*key)[16]) {
	unsigned char x[16] = { 0, }, xc[16] = { 0, };
	for (int i = 0; i < 16; ++i) {
		x[i] = in[i] ^ key[0][i];
	}
	for (int j = 1; j <= 10; ++j) {
		SubBytes(x, 16, 0);
		ShiftRows(x);
		if (j >= 10) {
			for (int i = 0; i < 16; ++i) {
				out[i] = x[i] ^ key[j][i];
			}
			break;
		}
		MixColumns(x, xc, 0);
		for (int i = 0; i < 16; ++i) {
			x[i] = xc[i] ^ key[j][i];
		}
	}
}

void make_dec(unsigned char *in, unsigned char *out, unsigned char(*key)[16]) {
	unsigned char x[16] = { 0, }, xc[16] = { 0, };
	for (int i = 0; i < 16; ++i) {
		x[i] = in[i] ^ key[10][i];
	}
	for (int j = 9; j >= 0; --j) {
		ShiftRowsR(x);
		SubBytes(x, 16, 1);
		for (int i = 0; i < 16; ++i) {
			xc[i] = x[i] ^ key[j][i];
		}
		if (j < 1) {
			for (int i = 0; i < 16; ++i) {
				out[i] = xc[i];
			}
			break;
		}
		MixColumns(xc, x, 1);
	}
}

	for (int i = 0; i < 16; ++i) {	//d4 27 11 ae e0 bf 98 f1 b8 b4 5d e5 1e 41 52 30
		r[i] = io[c % 16];	//0  13 10 7  4  1  14 11 8  5  2  15 12 9  6  3
		c += 13;

	for (int i = 0; i < 16; ++i) {
		io[i] = r[i];
	}
}

void MixColumns(unsigned char *in, unsigned char *out, int inv) {
	int a = 0;
	for (int i = 0; i < 4; ++i) {
		out[a] = xtime(in[a], C[inv][0]) ^ xtime(in[a + 1], C[inv][1]) ^ xtime(in[a + 2], C[inv][2]) ^ xtime(in[a + 3], C[inv][3]);

unsigned char xtime(unsigned char in1, unsigned char in2) {
	unsigned char gf8 = 0x1B;
	unsigned char check = 0x80;
	unsigned char bc = 0x01;
	unsigned char r = 0x00;
	for (int i = 0; i <= 7; ++i) {
		if (in2 & bc)
			r ^= in1;
		if (in1 & check)
			in1 = (in1 << 1) ^ gf8;
		else
			in1 <<= 1;
		bc <<= 1;

	}
	return r;
}

		r[i] = io[c % 16];	//0  5  10 15 4  9  14 3  8  13 2  7  12 1  6  11

}

void make_key(unsigned char *in, unsigned char (*out)[16]) {
	unsigned char b[16] = { 0, };
	int rcon = 0;

	for (int i = 0; i < 16; ++i) {
		out[0][i] = in[i];
	}
	for (int i = 0; i < 10; ++i) {
		Block(out[i], b, rcon);
		for (int j = 0; j < 16; ++j) {
			out[i + 1][j] = b[j];
		}
		rcon += 4;
	}
}

void Block(unsigned char *in, unsigned char *out, int rcon) {
	unsigned char x[4] = { 0, };
	unsigned char y[4] = { 0, };
	unsigned char z[4] = { 0, };
	int q = 11, w = 4;
	int i = 0, j = 0;

	for (; i <= 3; ++i) {
		x[i] = in[i];
	}
	for (i = 0; i <= 3; ++i) {
		++q;
		y[i] = in[q];
	}
	q = 4;
	RotWord(y);
	SubBytes(y, 4, 0);
	for (i = 0; i <= 3; i++) {
		z[i] = x[i] ^ y[i] ^ Rcon[rcon];
		++rcon;
	}
	for (i = 0; i <= 3; i++) {
		out[i] = z[i];
	}
	for (i = 0; i <= 2; ++i) {
		for (j = 0; j <= 3; j++) {
			x[j] = in[q];
			y[j] = z[j];
			++q;
		}
		for (j = 0; j <= 3; ++j) {
			z[j] = x[j] ^ y[j];
		}
		for (j = 0; j <= 3; j++) {
			out[j + w] = z[j];
		}
		w += 4;
	}
}

void SubBytes(unsigned char *io, int c, int inv) {
	for (int i = 0; i < c; ++i) {
		io[i] = S[inv][io[i]];
	}
}

void RotWord(unsigned char *io) {
	unsigned char a = io[0];
	for (int i = 0; i < 3; ++i) {
		io[i] = io[i + 1];
	}
	io[3] = a;
}
